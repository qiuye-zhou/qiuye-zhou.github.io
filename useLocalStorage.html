# LocalStorage

# 使用

import { useLocalStorage } from 'use-aid'

const uselocal = useLocalStorage()

uselocal.set('storage1', { 'name': 'aa', 'age': 18 })
const val = uselocal.get('storage1')

uselocal.removeAll()
